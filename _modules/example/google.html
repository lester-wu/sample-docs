"""Example Google style docstrings.

This module demonstrates documentation as specified by the `Google Python Style Guide`_.
Docstrings may extend over multiple lines. Sections are created
with a section header and a colon followed by a block of indented text.

Example:
    Examples can be given using either the `Example` or `Examples`
    sections. Sections support any reStructuredText formatting, including
    literal blocks::

        $ python example_google.py

Section breaks are created by resuming unindented text. Section breaks
are also implicitly created anytime a new section starts. There are a bunch
of sections providing by `napoleon` extension.

.. _Google Python Style Guide: http://google.github.io/styleguide/pyguide.html
"""

    def readwrite_property(self) -> typing.List[str]:

[docs]class ExampleError(Exception):
    """Exceptions are documented in the same way as classes.

    The __init__ method may be documented in either the class level
    docstring, or as a docstring on the __init__ method itself.

    Either form is acceptable, but the two should not be mixed. Choose one
    convention to document the __init__ method and be consistent with it.

    Note:
        Do not include the `self` parameter in the `Args` section.
    """

    def __init__(self, msg: str, code: str):
        self.msg = msg
        self.code = code




[docs]def example_generator(n: int) -> typing.Generator[str, None, None]:
    """Generators have a `Yields` section instead of a `Returns` section.

    Use the literal block `::` to introduce a code block. The literal block
    must be indented.

    Example:
        A description for the example::

            print([i for i in example_generator(4)])
            [0, 1, 2, 3]

    If you want to doctest the code block, please lead with `>>>`. It will be
    doctested when running `python -m doctest` command.

    Example:
        A description for the example(doctest)::

            >>> print([i for i in example_generator(4)])
            [0, 1, 2, 3]
    """
    for i in range(n):
        yield i
